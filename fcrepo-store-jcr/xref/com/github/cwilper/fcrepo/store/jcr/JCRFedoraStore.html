13  /**
14   * JCR-based implementation of {@link FedoraStore}.
15   * <p>
16   * In order to scale effectively to millions of objects, this implementation
17   * stores serialized objects and managed datastream content within a
18   * multi-level directory (<code>nt:folder</code>) structure based on the pid
19   * of the object. The first four lowercase hex digits of the md5 digest of the
20   * pid are used to compose the top and mid-level folder paths, and the pid
21   * with an underscore instead of a colon is used as the final folder path
22   * element, as follows (for object <code>test:o1</code>):
23   * <ul>
24   *   <li> <code>/08/cf/test_o1</code></li>
25   * </ul>
26   * <p>
27   * The serialized object is saved within an <code>nt:file</code> sub-node
28   * named <code>object</code>, which has an <code>nt:resource</code> sub-node
29   * named <code>jcr:content</code>, whose <code>jcr:data</code> property holds
30   * the binary value. So the serialized object <code>test:o1</code> would be
31   * found at:
32   * <ul>
33   *   <li> <code>/08/cf/test_o1/object/jcr:content[jcr:data]</code></li>
34   * </ul>
35   * <p>
36   * In addition, managed datastream content resides beneath a sub-folder of the
37   * object named using the datastream id. Within that folder, the content is
38   * stored in an <code>nt:file</code> sub-node named using the datastream
39   * version id, which has an <code>nt:resource</code> sub-node named
40   * <code>jcr:content</code>, whose <code>jcr:data</code> property holds
41   * the binary value. So the content of <code>test:o1</code>'s managed
42   * datastream with id <code>DS1</code> and version id <code>DS1.0</code>
43   * would be found at:
44   * <p>
45   * <ul>
46   *   <li> <code>/08/cf/test_o1/DS1/DS1.0/jcr:content[jcr:data]</code></li>
47   * </ul>
48   */
49  public class JCRFedoraStore implements FedoraStore {
50      private final Repository repository;
51      private final Credentials credentials;
52      private final DTOReader readerFactory;
53      private final DTOWriter writerFactory;
54  
55      /**
56       * Creates an instance.
57       *
58       * @param repository the repository to use.
59       * @param credentials the credentials to use when opening sessions.
60       * @param readerFactory the factory to use when objects need to be read.
61       * @param writerFactory the factory to use when objects need to be written.
62       * @throws NullPointerException if any argument is null.
63       */
64      public JCRFedoraStore(Repository repository, Credentials credentials,
65              DTOReader readerFactory, DTOWriter writerFactory) {
66          if (repository == null || credentials == null || readerFactory == null
67                  || writerFactory == null)
68              throw new NullPointerException();
69          this.repository = repository;
70          this.credentials = credentials;
71          this.readerFactory = readerFactory;
72          this.writerFactory = writerFactory;
73      }
74  
75      @Override
76      public FedoraStoreSession getSession() {
77          try {
78              return new JCRFedoraStoreSession(repository.login(credentials),
79                      readerFactory, writerFactory);
80          } catch (RepositoryException e) {
81              throw new StoreException("Error getting JCR session", e);
82          }
83      }
84  }
