26      public static boolean hasManagedDatastreamVersion(FedoraObject object,
27              String datastreamId, String datastreamVersionId) {
28          Datastream datastream = object.datastreams().get(datastreamId);
29          return datastream != null &&
30                  datastream.controlGroup() == ControlGroup.MANAGED &&
31                  hasDatastreamVersion(datastream, datastreamVersionId);
32      }
33  
34      public static boolean hasDatastreamVersion(Datastream datastream,
35              String datastreamVersionId) {
36          for (DatastreamVersion version : datastream.versions()) {
37              if (version.id().equals(datastreamVersionId)) return true;
38          }
39          return false;
40      }
41  
42      public static void closeOrWarn(Closeable stream) {
43          try {

45          } catch (IOException e) {
46              logger.warn(CommonConstants.ERR_CLOSING_STREAM, e);
