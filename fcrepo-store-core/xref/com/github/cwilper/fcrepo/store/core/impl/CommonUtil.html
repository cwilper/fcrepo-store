24      public static String getDetails(String pid,
25              String datastreamId, String datastreamVersionId) {
26          return "(pid=" + pid + ", datastreamId=" + datastreamId
27                  + ", datastreamVersionId=" + datastreamVersionId + ")";
28      }

30      public static boolean hasManagedDatastreamVersion(FedoraObject object,
31              String datastreamId, String datastreamVersionId) {
32          Datastream datastream = object.datastreams().get(datastreamId);
33          return datastream != null &&
34                  datastream.controlGroup() == ControlGroup.MANAGED &&
35                  hasDatastreamVersion(datastream, datastreamVersionId);
36      }
37  
38      public static boolean hasDatastreamVersion(Datastream datastream,
39              String datastreamVersionId) {
40          for (DatastreamVersion version : datastream.versions()) {
41              if (version.id().equals(datastreamVersionId)) return true;
42          }
43          return false;
44      }
45  
46      public static void closeOrWarn(Closeable stream) {
47          try {
48              if (stream != null) stream.close();
49          } catch (IOException e) {
50              logger.warn(CommonConstants.ERR_CLOSING_STREAM, e);
51          }
52      }
53  
54      public static void writeObject(DTOWriter writerFactory,
55              FedoraObject object, OutputStream outputStream)
56              throws IOException {
57          DTOWriter writer = writerFactory.getInstance();
58          boolean success = false;
59          try {
60              writer.writeObject(object, outputStream);
61              outputStream.close();
62              success = true;
63          } finally {
64              writer.close();
65              if (!success) closeOrWarn(outputStream);
66          }
67      }
68  
69      public static FedoraObject readObject(DTOReader readerFactory,
70              InputStream inputStream) throws IOException {
71          DTOReader reader = readerFactory.getInstance();
72          try {
73              return reader.readObject(inputStream);
74          } finally {
75              reader.close();
76          }
77      }
