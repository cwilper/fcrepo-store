42      AkubraFedoraStoreSession(BlobStore objectStore, BlobStore contentStore,
43              DTOReader readerFactory, DTOWriter writerFactory) {
44          if (objectStore == null || contentStore == null
45                  || readerFactory == null || writerFactory == null) {
46              throw new NullPointerException();
47          }
48          this.objectStore = objectStore;
49          this.contentStore = contentStore;
50          this.readerFactory = readerFactory;
51          this.writerFactory = writerFactory;
52          this.closed = false;

55      @Override
56      public void addObject(FedoraObject object) {
57          ensureNotClosed();
58          if (object == null) throw new NullPointerException();
59          if (object.pid() == null) throw new IllegalArgumentException();
60          try {
61              Blob blob = Util.getBlob(getObjectStoreConnection(), object.pid());
62              if (blob.exists()) throw new ExistsException(object.pid());
63              Util.writeObject(writerFactory, object, blob);
64          } catch (IOException e) {
65              throw new StoreException(Constants.ERR_ADDING_OBJ, e);
66          }
67      }

69      @Override
70      public FedoraObject getObject(String pid) {
71          ensureNotClosed();
72          if (pid == null) throw new NullPointerException();
73          try {
74              Blob blob = Util.getBlob(getObjectStoreConnection(), pid);
75              if (!blob.exists()) throw new NotFoundException(
76                      Constants.ERR_NOTFOUND_OBJ_IN_STORAGE + ": " + pid);
77              return Util.readObject(readerFactory, blob);
78          } catch (IOException e) {
79              throw new StoreException(Constants.ERR_GETTING_OBJ, e);
80          }
81      }

83      @Override
84      public void updateObject(FedoraObject object) {
85          ensureNotClosed();
86          if (object == null) throw new NullPointerException();
87          if (object.pid() == null) throw new IllegalArgumentException();
88          try {
89              Blob blob = Util.getBlob(getObjectStoreConnection(), object.pid());
90              if (!blob.exists()) throw new NotFoundException(
91                      Constants.ERR_NOTFOUND_OBJ_IN_STORAGE + ": "
92                      + object.pid());
93              deleteOldManagedContent(
94                      Util.readObject(readerFactory, blob), object);
95              Util.writeObject(writerFactory, object, blob);
96          } catch (IOException e) {
97              throw new StoreException(Constants.ERR_UPDATING_OBJ, e);
98          }
99      }

144     @Override
145     public void setContent(String pid, String datastreamId,
146             String datastreamVersionId, InputStream inputStream) {
147         ensureNotClosed();
148         if (inputStream == null) throw new NullPointerException();
149         Blob blob = getContentBlob(
150                 pid, datastreamId, datastreamVersionId, false);
151         OutputStream outputStream = null;
152         boolean success = false;
153         try {
154             outputStream = blob.openOutputStream(-1, true);
155             IOUtils.copyLarge(inputStream, outputStream);
156             inputStream.close();
157             outputStream.close();
158             success = true;
159         } catch (IOException e) {
160             throw new StoreException(Constants.ERR_SETTING_CONT, e);
161         } finally {
162             if (!success) {
163                 Util.closeOrWarn(inputStream);
164                 Util.closeOrWarn(outputStream);
165             }
166         }
167     }

169     @Override
170     public void close() {
171         if (!closed) {
172             try {
173                 if (objectStoreConnection != null) {
174                     objectStoreConnection.close();
175                 }
176                 if (objectStoreConnection != null) {
177                     objectStoreConnection.close();
178                 }
179             } finally {
180                 closed = true;
181             }
182         }

198     private Blob getContentBlob(String pid, String datastreamId,
199             String datastreamVersionId, boolean mustExist) {
200         if (pid == null  || datastreamId == null ||
201                 datastreamVersionId == null) throw new NullPointerException();
202         FedoraObject object = getObject(pid);
203         if (!Util.hasManagedDatastreamVersion(object, datastreamId,
204                 datastreamVersionId)) {
205             throw new NotFoundException(Constants.ERR_NOTFOUND_DS_IN_OBJ + " "
206                     + Util.getDetails(pid, datastreamId, datastreamVersionId));
207         }
208         try {
209             Blob blob = Util.getBlob(getContentStoreConnection(),
210                     pid, datastreamId, datastreamVersionId);
211             if (mustExist && !blob.exists()) {
212                 throw new NotFoundException(
213                         Constants.ERR_NOTFOUND_DS_IN_STORAGE + " "
214                         + Util.getDetails(pid, datastreamId,
215                         datastreamVersionId));
216             }
217             return blob;
218         } catch (IOException e) {
219             throw new StoreException(Constants.ERR_GETTING_CONT, e);
220         }
221     }

225             String datastreamVersionId) {
226         try {
227             Util.getBlob(getContentStoreConnection(), pid, datastreamId,
228                     datastreamVersionId).delete();
229         } catch (Exception e) {

231                     pid, datastreamId, datastreamVersionId), e);
232         }

235     // if newObject is null, all managed content will be deleted
236     private void deleteOldManagedContent(FedoraObject oldObject,
237             FedoraObject newObject) {
238         for (Datastream datastream : oldObject.datastreams().values()) {
239             if (datastream.controlGroup() == ControlGroup.MANAGED) {
240                 String datastreamId = datastream.id();
241                 for (DatastreamVersion datastreamVersion : datastream.versions()) {
242                     String datastreamVersionId = datastreamVersion.id();
243                     if (newObject == null ||
244                             !Util.hasManagedDatastreamVersion(
245                             newObject, datastreamId, datastreamVersionId)) {
246                         deleteContent(oldObject.pid(), datastreamId,
247                             datastreamVersionId);
248                     }
249                 }
250             }
251         }
252     }
