14  /**
15   * An iterator of {@link FedoraObject}s that automatically closes a given
16   * {@link BlobStoreConnection} when finished.
17   */
18  class ConnectionClosingObjectIterator
19          extends AbstractIterator<FedoraObject> {
20      private static final Logger logger =
21              LoggerFactory.getLogger(ConnectionClosingObjectIterator.class);
22  
23      private final BlobStoreConnection connection;
24      private final Iterator<URI> ids;
25      private final DTOReader readerFactory;
26  
27      /**
28       * Creates an instance.
29       *
30       * @param connection the connection.
31       * @param ids the blob ids of the Fedora objects to iterate.
32       * @param readerFactory the factory to use to deserialize the objects.
33       */
34      public ConnectionClosingObjectIterator(BlobStoreConnection connection,
35              Iterator<URI> ids, DTOReader readerFactory) {
36          this.connection = connection;
37          this.ids = ids;
38          this.readerFactory = readerFactory;
39      }
40  
41      @Override
42      protected FedoraObject computeNext() {
43          while (ids.hasNext()) {
44              URI id = ids.next();
45              try {
46                  return Util.readObject(readerFactory,
47                          connection.getBlob(id, null));
48              } catch (IOException e) {
49                  logger.warn(Constants.ERR_PARSING_OBJ + ": " + id, e);
50              }
51          }
52          connection.close();
53          return endOfData();
54      }
55  
56      @Override
57      protected void finalize() throws Throwable {
58          try {
59              super.finalize();
60          } finally {
61              connection.close();
62          }
63      }
64  }
