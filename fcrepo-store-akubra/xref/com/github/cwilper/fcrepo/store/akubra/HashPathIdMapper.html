11  /**
12   * Provides a hash-based <code>file:</code> mapping for any URI.
13   * <p>
14   * The path component of each internal URI is derived from an MD5 hash of
15   * the external URI. The filename component is a reversible encoding of the
16   * external URI that is safe to use as a filename on modern filesystems.
17   * <p>
18   * <h2>Hash Path Patterns</h2>
19   * The pattern given at construction time determines how the path component
20   * of each internal URI will be composed. Within the pattern, the # character
21   * is a stand-in for a hexadecimal [0-f] digit from the MD5 hash of the
22   * external id.
23   * <p>
24   * Patterns:
25   * <ul>
26   *   <li> must consist only of # and / characters.</li>
27   *   <li> must contain between 1 and 32 # characters.</li>
28   *   <li> must not begin or end with the / character.</li>
29   *   <li> must not contain consecutive / characters.</li>
30   * </ul>
31   * <p>
32   * Example patterns:
33   * <ul>
34   *   <li> Good: #</li>
35   *   <li> Good: ##/#</li>
36   *   <li> Good: ##/##/##</li>
37   *   <li> Bad: a</li>
38   *   <li> Bad: ##/</li>
39   *   <li> Bad: ##//##</li>
40   * </ul>
41   * <p>
42   * <h2>Filesystem-Safe Encoding</h2>
43   * The last part of the internal URI is a "filesystem-safe" encoding of the
44   * external URI. All characters will be UTF-8 percent-encoded ("URI escaped")
45   * except for the following: <code>a-z A-Z 0-9 = ( ) [ ] -</code>
46   * In addition, <code>.</code> (period) will be escaped as <code>%2E</code> when
47   * it occurs as the last character of the URI.
48   * <p>
49   * <h2>Example Mappings</h2>
50   * With pattern <em>#/#</em>:
51   * <ul>
52   *   <li> <code>urn:example1</code> becomes <code>file:0/8/urn%3Aexample1</code></li>
53   *   <li> <code>http://tinyurl.com/cxzzf</code> becomes <code>file:6/2/http%3A%2F%2Ftinyurl.com%2Fcxzzf</code></li>
54   * </ul>
55   * With pattern <em>##/##</em>:
56   * <ul>
57   *   <li> <code>urn:example1</code> becomes <code>file:08/86/urn%3Aexample1</code></li>
58   *   <li> <code>http://tinyurl.com/cxzzf</code> becomes <code>file:62/ca/http%3A%2F%2Ftinyurl.com%2Fcxzzf</code></li>
59   * </ul>
60   */
61  public class HashPathIdMapper
62          implements IdMapper {

66      private final String pattern;

68      /**
69       * Creates an instance that will use the given pattern.
70       *
71       * @param pattern the pattern to use, possibly <code>null</code> or "".
72       * @throws IllegalArgumentException if the pattern is invalid.
73       */
74      public HashPathIdMapper(String pattern) {
75          this.pattern = validatePattern(pattern);
76      }

78      //@Override
79      public URI getExternalId(URI internalId) throws NullPointerException {
80          String fullPath = internalId.toString().substring(
81                  internalScheme.length() + 1);
82          int i = fullPath.lastIndexOf('/');
83          String encodedURI;
84          if (i == -1)
85              encodedURI = fullPath;
86          else
87              encodedURI = fullPath.substring(i + 1);
88          return URI.create(decode(encodedURI));
89      }
90  
91      //@Override
92      public URI getInternalId(URI externalId) throws NullPointerException {
93          if (externalId == null) {
94              throw new NullPointerException();
95          }
96          String uri = externalId.toString();
97          return URI.create(internalScheme + ":" + getPath(uri) + encode(uri));
98      }

100     //@Override
101     public String getInternalPrefix(String externalPrefix)
102             throws NullPointerException {
103         if (externalPrefix == null) {
104             throw new NullPointerException();
105         }
106         // we can only do this if pattern is ""
107         if (pattern.length() == 0) {
108             return internalScheme + ":" + encode(externalPrefix);
109         } else {
110             return null;
111         }
112     }

114     // gets the path based on the hash of the uri, or "" if the pattern is empty
115     private String getPath(String uri) {
116         if (pattern.length() == 0) {
117             return "";
118         }
119         StringBuilder builder = new StringBuilder();
120         String hash = DigestUtils.md5Hex(uri);
121         int hashPos = 0;
122         for (int i = 0; i < pattern.length(); i++) {
123             char c = pattern.charAt(i);
124             if (c == '#') {
125                 builder.append(hash.charAt(hashPos++));
126             } else {
127                 builder.append(c);
128             }
129         }
130         builder.append('/');
131         return builder.toString();
132     }

134     private static String encode(String uri) {
135         // encode char-by-char because we only want to borrow
136         // URLEncoder.encode's behavior for some characters
137         StringBuilder out = new StringBuilder();
138         for (int i = 0; i < uri.length(); i++) {
139             char c = uri.charAt(i);
140             if (c >= 'a' && c <= 'z') {
141                 out.append(c);
142             } else if (c >= '0' && c <= '9') {
143                 out.append(c);
144             } else if (c >= 'A' && c <= 'Z') {
145                 out.append(c);
146             } else if (c == '-' || c == '=' || c == '(' || c == ')'
147                     || c == '[' || c == ']' || c == ';') {
148                 out.append(c);
149             } else if (c == ':') {
150                 out.append("%3A");
151             } else if (c == ' ') {
152                 out.append("%20");
153             } else if (c == '+') {
154                 out.append("%2B");
155             } else if (c == '_') {
156                 out.append("%5F");
157             } else if (c == '*') {
158                 out.append("%2A");
159             } else if (c == '.') {
160                 if (i == uri.length() - 1) {
161                     out.append("%2E");
162                 } else {
163                     out.append(".");
164                 }
165             } else {
166                 try {
167                     out.append(URLEncoder.encode("" + c, "UTF-8"));
168                 } catch (UnsupportedEncodingException wontHappen) {
169                     throw new RuntimeException(wontHappen);
170                 }
171             }
172         }
173         return out.toString();
174     }

176     private static String decode(String encodedURI) {
177         if (encodedURI.endsWith("%2E")) {
178             encodedURI = encodedURI.substring(0, encodedURI.length() - 3) + ".";
179         }
180         try {
181             return URLDecoder.decode(encodedURI, "UTF-8");
182         } catch (UnsupportedEncodingException wontHappen) {
183             throw new RuntimeException(wontHappen);
184         }
185     }
186 
187     private static String validatePattern(String pattern) {
188         if (pattern == null) {
189             return "";
190         }
191         int count = 0;
192         boolean prevWasSlash = false;
193         for (int i = 0; i < pattern.length(); i++) {
194             char c = pattern.charAt(i);
195             if (c == '#') {
196                 count++;
197                 prevWasSlash = false;
198             } else if (c == '/') {
199                 if (i == 0 || i == pattern.length() - 1) {
200                     throw new IllegalArgumentException("Pattern must not begin"
201                             + " or end with '/'");
202                 } else if (prevWasSlash) {
203                     throw new IllegalArgumentException("Pattern must not"
204                             + " contain consecutive '/' characters");
205                 } else {
206                     prevWasSlash = true;
207                 }
208             } else {
209                 throw new IllegalArgumentException("Illegal character in"
210                         + " pattern: " + c);
211             }
212         }
213         if (count > 32) {
214             throw new IllegalArgumentException("Pattern must not contain more"
215                     + " than 32 '#' characters");
216         }
217         return pattern;
218     }
