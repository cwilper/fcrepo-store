31      static void writeObject(DTOWriter writerFactory,
32              FedoraObject object, Blob blob) throws IOException {
33          DTOWriter writer = writerFactory.getInstance();
34          OutputStream out = blob.openOutputStream(-1, true);
35          boolean success = false;
36          try {
37              writer.writeObject(object, out);
38              out.close();
39              success = true;
40          } finally {
41              writer.close();
42              if (!success) closeOrWarn(out);
43          }
44      }

46      static FedoraObject readObject(DTOReader readerFactory,
47              Blob blob) throws IOException {
48          DTOReader reader = readerFactory.getInstance();
49          try {
50              return reader.readObject(blob.openInputStream());
51          } finally {
52              reader.close();
53          }
54      }
