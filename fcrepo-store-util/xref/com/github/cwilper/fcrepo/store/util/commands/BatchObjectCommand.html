10  /**
11   * Base class for {@link Command}s that work with multiple
12   * {@link FedoraObject}s in a {@link com.github.cwilper.fcrepo.store.core.FedoraStoreSession}.
13   */
14  public abstract class BatchObjectCommand implements Command {
15      private static final Logger logger =
16              LoggerFactory.getLogger(BatchObjectCommand.class);
17  
18      protected final FedoraStoreSession source;
19      protected final IdSpec pids;
20  
21      public BatchObjectCommand(FedoraStoreSession source, IdSpec pids) {
22          this.source = source;
23          this.pids = pids;
24      }
25      
26      @Override
27      public void execute() {
28          if (pids.isDynamic()) {
29              for (FedoraObject object : source) {
30                  if (pids.matches(object.pid())) {
31                      handleObject(object);
32                  } else {
33                      logger.debug("Skipped {} (pid filtered out)",
34                              object.pid());
35                  }
36              }
37          } else {
38              for (String pid : pids) {
39                  FedoraObject object = null;
40                  try {
41                      object = source.getObject(pid);
42                  } catch (NotFoundException e) {
43                      logger.warn("Skipped {} (not in source)", pid);
44                  }
45                  if (object != null) handleObject(object);
46              }
47          }
48      }
49      
50      @Override
51      public void close() {
52          source.close();
53      }
54  
55      protected abstract void handleObject(FedoraObject object);
56  }
