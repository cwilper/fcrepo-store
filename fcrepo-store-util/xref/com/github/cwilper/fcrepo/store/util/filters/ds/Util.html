34      static InputStream getInputStream(String info, String pid, Datastream ds,
35              DatastreamVersion dsv, ContentResolver contentResolver,
36              String localFedoraServer) throws IOException {
37          try {
38              if (ds.controlGroup() == ControlGroup.INLINE_XML) {
39                  if (!dsv.inlineXML().canonical()) {
40                      try {
41                          XMLUtil.canonicalize(dsv.inlineXML().bytes());
42                      } catch (IOException e) {
43                          logger.warn("Unable to canonicalize {} using C14N11;"
44                                  + " using non-standard method ("
45                                  + e.getCause().getMessage() + ")", info);
46                      }
47                  }
48                  return new ByteArrayInputStream(dsv.inlineXML().bytes());
49              } else if (ds.controlGroup() == ControlGroup.MANAGED) {
50                  return CommandContext.getSource().getContent(pid,
51                          ds.id(), dsv.id());
52              } else {
53                  String location = dsv.contentLocation().toString();
54                  location = location.replace("local.fedora.server",
55                          localFedoraServer);
56                  return contentResolver.resolveContent(null,
57                          URI.create(location));
58              }
59          } catch (StoreException e) {
60              throw new IOException(e);
61          }
62      }

64      static String[] computeFixity(InputStream inputStream,
65              String algorithm) throws IOException {
66          try {
67              MessageDigest digest = MessageDigest.getInstance(algorithm);
68              byte[] buffer = new byte[4096];
69              int count;
70              long size = 0;
71              while ((count = inputStream.read(buffer)) > 0) {
72                  size += count;
73                  digest.update(buffer, 0, count);
74              }
75              return new String[] { "" + size, hexString(digest.digest()) };
76          } catch (NoSuchAlgorithmException e) {
77              throw new IOException(e);
78          } finally {
79              IOUtils.closeQuietly(inputStream);
80          }
81      }

83      static final String HEXES = "0123456789abcdef";
84      static String hexString(byte[] bytes) {
85          StringBuilder hex = new StringBuilder(2 * bytes.length);
86          for (byte b : bytes) {
87              hex.append(HEXES.charAt((b & 0xF0) >> 4))
88                      .append(HEXES.charAt((b & 0x0F)));
89          }
90          return hex.toString();
91      }

102     // create or update an existing object so it refers to the given
103     // managed datastream -- a prerequisite to putting the content into
104     // the store.
105     static void putObjectIfNoSuchManagedDatastream(FedoraObject object,
106             FedoraStoreSession store, String datastreamId) {
107         try {
108             FedoraObject existing = store.getObject(object.pid());
109             Datastream ds = existing.datastreams().get(datastreamId);
110             if (ds == null || ds.controlGroup() != ControlGroup.MANAGED) {
111                 store.updateObject(object);
112             }
113         } catch (NotFoundException e) {
114             store.addObject(object);
115         }
