3   import java.util.Calendar;
4   import java.util.GregorianCalendar;

6   /**
7    * Timestamp-based {@link PathAlgorithm} compatible with Fedora's legacy
8    * lowlevel storage path algorithm.
9    */
10  public class TimestampPathAlgorithm implements PathAlgorithm {
11  
12      @Override
13      public String generatePath(String id) {
14          return dateBasedPath(new GregorianCalendar()) + encode(id);
15      }
16  
17      @Override
18      public String getId(String path) {
19          return decode(path.substring(path.lastIndexOf("/") + 1));
20      }
21      
22      private static String dateBasedPath(GregorianCalendar date) {
23          return date.get(Calendar.YEAR) + "/"
24                  + pad(2, date.get(Calendar.MONTH))
25                  + pad(2, date.get(Calendar.DAY_OF_MONTH)) + "/"
26                  + pad(2, date.get(Calendar.HOUR_OF_DAY)) + "/"
27                  + pad(2, date.get(Calendar.MINUTE)) + "/";
28      }
29      
30      private static String encode(String in) {
31          int i = in.indexOf("+");
32          if (i == -1) return pidToFilename(in);
33          return pidToFilename(in.substring(0, i)) + in.substring(i);
34      }
35      
36      private static String decode(String in) {
37          int i = in.indexOf("+");
38          if (i == -1) return filenameToPid(in);
39          return filenameToPid(in.substring(0, i)) + in.substring(i);
40      }
41  
42      private static String pad(int requiredLength, int in) {
43          String out = Integer.toString(in);
44          while (out.length() < requiredLength) {
45              out = "0" + out;
46          }
47          return out;
48      }
49      
50      private static String pidToFilename(String pid) {
51          String filename = pid.replaceFirst(":", "_");
52          if (filename.endsWith(".")) {
53              return filename.substring(0, filename.length() - 1) + "%";
54          }
55          return filename;
56      }
57      
58      private static String filenameToPid(String filename) {
59          String pid = filename.replaceFirst("_", ":");
60          if (pid.endsWith("%")) {
61              return pid.substring(0, pid.length() - 1) + ".";
62          }
63          return pid;
64      }
