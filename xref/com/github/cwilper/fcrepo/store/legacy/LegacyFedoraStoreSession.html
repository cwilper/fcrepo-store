25   * Legacy {@link FedoraStoreSession} implementation compatible with pre-Akubra

28  class LegacyFedoraStoreSession implements FedoraStoreSession {
29      private static final Logger logger =
30              LoggerFactory.getLogger(LegacyFedoraStoreSession.class);

39      LegacyFedoraStoreSession(FileStore objectStore, FileStore contentStore,
40              DTOReader readerFactory, DTOWriter writerFactory) {
41          if (objectStore == null || contentStore == null
42                  || readerFactory == null || writerFactory == null) {
43              throw new NullPointerException();
44          }
45          this.objectStore = objectStore;
46          this.contentStore = contentStore;
47          this.readerFactory = readerFactory;
48          this.writerFactory = writerFactory;
49          this.closed = false;
50      }

52      @Override
53      public void addObject(FedoraObject object) {
54          ensureNotClosed();
55          if (object == null) throw new NullPointerException();
56          if (object.pid() == null) throw new IllegalArgumentException();
57          String path = objectStore.getPath(object.pid());
58          if (path != null) throw new ExistsException(object.pid());
59          path = objectStore.generatePath(object.pid());
60          objectStore.setPath(object.pid(), path);
61          boolean success = false;
62          try {
63              Util.writeObject(writerFactory, object,
64                      objectStore.getFileOutputStream(path));
65              success = true;
66          } catch (IOException e) {
67              throw new StoreException(CommonConstants.ERR_ADDING_OBJ, e);
68          } finally {
69              if (!success) {
70                  objectStore.setPath(object.pid(), null);
71              }
72          }
73      }

75      @Override
76      public FedoraObject getObject(String pid) {
77          ensureNotClosed();
78          if (pid == null) throw new NullPointerException();
79          try {
80              String path = objectStore.getPath(pid);
81              if (path == null) throw new NotFoundException(
82                      CommonConstants.ERR_NOTFOUND_OBJ_IN_STORAGE + ": " + pid);
83              return Util.readObject(readerFactory,
84                      objectStore.getFileInputStream(path));
85          } catch (IOException e) {
86              throw new StoreException(CommonConstants.ERR_GETTING_OBJ, e);
87          }
88      }

90      @Override
91      public void updateObject(FedoraObject object) {
92          ensureNotClosed();
93          if (object == null) throw new NullPointerException();
94          if (object.pid() == null) throw new IllegalArgumentException();
95          try {
96              String path = objectStore.getPath(object.pid());
97              if (path == null) throw new NotFoundException(
98                      CommonConstants.ERR_NOTFOUND_OBJ_IN_STORAGE + ": "
99                      + object.pid());
100             deleteOldManagedContent(Util.readObject(readerFactory,
101                     objectStore.getFileInputStream(path)), object);
102             Util.writeObject(writerFactory, object,
103                     objectStore.getFileOutputStream(path));
104         } catch (IOException e) {
105             throw new StoreException(CommonConstants.ERR_UPDATING_OBJ, e);
106         }
107     }

109     @Override
110     public void deleteObject(String pid) {
111         ensureNotClosed();
112         if (pid == null) throw new NullPointerException();
113         try {
114             String path = objectStore.getPath(pid);
115             if (path == null) throw new NotFoundException(
116                     CommonConstants.ERR_NOTFOUND_OBJ_IN_STORAGE + ": " + pid);
117             deleteOldManagedContent(
118                     Util.readObject(readerFactory,
119                             objectStore.getFileInputStream(path)), null);
120             objectStore.deleteFile(path);
121             objectStore.setPath(pid, null);
122         } catch (IOException e) {
123             throw new StoreException(CommonConstants.ERR_DELETING_OBJ, e);
124         }
125     }

145     @Override
146     public void setContent(String pid, String datastreamId,
147             String datastreamVersionId, InputStream inputStream) {
148         ensureNotClosed();
149         if (inputStream == null) throw new NullPointerException();
150         OutputStream outputStream = null;
151         boolean success = false;
152         try {
153             String path = getContentPath(
154                     pid, datastreamId, datastreamVersionId, false);
155             outputStream = contentStore.getFileOutputStream(path);
156             IOUtils.copyLarge(inputStream, outputStream);
157             inputStream.close();
158             outputStream.close();
159             success = true;
160         } catch (IOException e) {
161             throw new StoreException(CommonConstants.ERR_SETTING_CONT, e);
162         } finally {
163             if (!success) {
164                 Util.closeOrWarn(inputStream);
165                 Util.closeOrWarn(outputStream);
166             }
167         }
168     }

175     @Override
176     public Iterator<FedoraObject> iterator() {
177         ensureNotClosed();
178         final Iterator<String> paths = objectStore.iterator();
179         return new AbstractIterator<FedoraObject>() {
180             @Override
181             protected FedoraObject computeNext() {
182                 while (paths.hasNext()) {
183                     String path = paths.next();
184                     try {
185                         return Util.readObject(readerFactory,
186                                 objectStore.getFileInputStream(path));
187                     } catch (IOException e) {
188                         logger.warn(CommonConstants.ERR_PARSING_OBJ + ": "
189                                 + path, e);
190                     }
191                 }
192                 return endOfData();
193             }
194         };
195     }

197     private String getContentPath(String pid, String datastreamId,
198             String datastreamVersionId, boolean mustExist) {
199         if (pid == null  || datastreamId == null ||
200                 datastreamVersionId == null) throw new NullPointerException();
201         FedoraObject object = getObject(pid);
202         if (!Util.hasManagedDatastreamVersion(object, datastreamId,
203                 datastreamVersionId)) {
204             throw new NotFoundException(CommonConstants.ERR_NOTFOUND_DS_IN_OBJ + " "
205                     + Util.getDetails(pid, datastreamId, datastreamVersionId));
206         }
207         String id = Util.getId(pid, datastreamId, datastreamVersionId);
208         String path = contentStore.getPath(id);
209         if (path == null) {
210             if (mustExist) {
211                 throw new NotFoundException(
212                         CommonConstants.ERR_NOTFOUND_DS_IN_STORAGE + " "
213                         + Util.getDetails(pid, datastreamId,
214                         datastreamVersionId));
215             } else {
216                 path = contentStore.generatePath(id);
217                 contentStore.setPath(id, path);
218             }
219         }
220         return path;
221     }

223     // just log a warning message in the event of failure
224     private void deleteContent(String pid, String datastreamId,
225             String datastreamVersionId) {
226         String id = Util.getId(pid, datastreamId, datastreamVersionId);
227         String path = contentStore.getPath(id);
228         if (path == null) {
229             logger.warn(CommonConstants.ERR_DELETING_CONT + " " + Util.getDetails(
230                     pid, datastreamId, datastreamVersionId)
231                     + ": No such datastream in registry");
232         }
233         try {
234             contentStore.deleteFile(path);
235             contentStore.setPath(id, null);
236         } catch (Exception e) {
237             logger.warn(CommonConstants.ERR_DELETING_CONT + " " + Util.getDetails(
238                     pid, datastreamId, datastreamVersionId), e);
239         }
240     }

242     // if newObject is null, all managed content will be deleted
243     private void deleteOldManagedContent(FedoraObject oldObject,
244             FedoraObject newObject) {
245         for (Datastream ds : oldObject.datastreams().values()) {
246             if (ds.controlGroup() == ControlGroup.MANAGED) {
247                 String dsId = ds.id();
248                 for (DatastreamVersion dsv : ds.versions()) {
249                     String dsvId = dsv.id();
250                     if (newObject == null ||
251                             !Util.hasManagedDatastreamVersion(
252                                     newObject, dsId, dsvId)) {
253                         deleteContent(oldObject.pid(), dsId, dsvId);
254                     }
255                 }
256             }
257         }
258     }

261         if (closed) throw new IllegalStateException("Session is closed");
262     }
