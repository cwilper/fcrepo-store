9   /**
10   * Database-backed {@link PathRegistry} implementation.
11   */
12  public class DBPathRegistry implements PathRegistry {
13      private static final String CREATE_TABLE_DDL =
14              "CREATE TABLE ? (\n"
15              + "id VARCHAR(256) PRIMARY KEY NOT NULL,\n"
16              + "path VARCHAR(4096) NOT NULL)";
17      private static final String SELECT_COUNT_SQL =
18              "SELECT COUNT(*) FROM ?";
19      private static final String SELECT_PATH_SQL =
20              "SELECT PATH FROM ? WHERE ID = ?";
21      private static final String INSERT_PATH_SQL =
22              "INSERT INTO ? (id, path) VALUES (?, ?)";
23      private static final String UPDATE_PATH_SQL =
24              "UPDATE ? SET path = ? WHERE id = ?";
25      private static final String DELETE_BY_ID_SQL =
26              "DELETE FROM ? WHERE id = ?";
27      private static final String DELETE_ALL_SQL =
28              "DELETE FROM ?";
29      
30  
31      private final JdbcTemplate db;
32      private final String table;
33  
34      /**
35       * Creates an instance.
36       *
37       * @param db the database to use.
38       * @param table the name of the table, which will be created if it doesn't
39       *              exist.
40       * @throws NullPointerException if either argument is null.
41       */
42      public DBPathRegistry(JdbcTemplate db, String table) {
43          if (db == null || table == null) throw new NullPointerException();
44          this.db = db;
45          this.table = table;
46          createTableIfNeeded();
47      }
48  
49      private void createTableIfNeeded() {
50          try {
51              getPathCount();
52          } catch (StoreException e) {
53              try {
54                  db.execute(CREATE_TABLE_DDL.replace("?", table));
55                  getPathCount();
56              } catch (Exception e2) {
57                  throw new StoreException("Error creating table", e2);
58              }
59          }
60      }
61  
62      /**
63       * Deletes all rows from the table, for testing.
64       */
65      public void clear() {
66          try {
67              db.update(DELETE_ALL_SQL.replace("?", table));
68          } catch (DataAccessException e) {
69              throw new StoreException("Error clearing table", e);
70          }
71      }
72  
73      @Override
74      public long getPathCount() {
75          try {
76              return db.queryForLong(
77                      SELECT_COUNT_SQL.replaceFirst("\\?", table));
78          } catch (DataAccessException e) {
79              throw new StoreException("Error getting path count", e);
80          }
81      }
82  
83      @Override
84      public String getPath(String id) {
85          try {
86              List<String> result = db.queryForList(
87                      SELECT_PATH_SQL.replaceFirst("\\?", table),
88                      new String[] { id }, String.class);
89              if (result.size() == 0) return null;
90              return result.get(0);
91          } catch (DataAccessException e) {
92              throw new StoreException("Error getting path", e);
93          }
94      }
95  
96      @Override
97      public void setPath(String id, String path) {
98          boolean exists = getPath(id) != null;
99          try {
100             if (!exists && path != null) {
101                 db.update(INSERT_PATH_SQL.replaceFirst("\\?", table),
102                         id, path);
103             } else if (exists) {
104                 if (path != null) {
105                     db.update(UPDATE_PATH_SQL.replaceFirst("\\?", table),
106                             path, id);
107                 } else {
108                     db.update(DELETE_BY_ID_SQL.replaceFirst("\\?", table),
109                             id);
110                 }
111             }
112         } catch (DataAccessException e) {
113             throw new StoreException("Error setting path", e);
114         }
115     }
