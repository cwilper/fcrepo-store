14  /**
15   * An iterator of {@link FedoraObject}s in an Akubra blob store.
16   */
17  class AkubraObjectIterator
18          extends AbstractIterator<FedoraObject> {
19      private static final Logger logger =
20              LoggerFactory.getLogger(AkubraObjectIterator.class);
21  
22      private final BlobStoreConnection connection;
23      private final Iterator<URI> ids;
24      private final DTOReader readerFactory;
25  
26      /**
27       * Creates an instance.
28       *
29       * @param connection the connection.
30       * @param ids the blob ids of the Fedora objects to iterate.
31       * @param readerFactory the factory to use to deserialize the objects.
32       */
33      public AkubraObjectIterator(BlobStoreConnection connection,
34              Iterator<URI> ids, DTOReader readerFactory) {
35          this.connection = connection;
36          this.ids = ids;
37          this.readerFactory = readerFactory;
38      }
39  
40      @Override
41      protected FedoraObject computeNext() {
42          while (ids.hasNext()) {
43              URI id = ids.next();
44              try {
45                  return Util.readObject(readerFactory,
46                          connection.getBlob(id, null));
47              } catch (IOException e) {
48                  logger.warn(Constants.ERR_PARSING_OBJ + ": " + id, e);
49              }
50          }
51          return endOfData();
52      }
53  }
